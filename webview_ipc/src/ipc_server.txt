#[derive(Debug, Serialize, Deserialize, Clone)]
pub struct CmdMessage {
    pub cmd: String,
    pub message: String,
}
pub fn create_ipc_name() -> String {
    let nanos = SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .unwrap()
        .as_nanos();

    // Ubah ke bentuk Hexadecimal (String)
    let random_string = format!("myipc-{:x}", nanos);

    random_string
}

async fn handle_client(mut stream: Stream, mapfun: ArcCallBack) {
    let mut buffer = vec![0; 1024 * 1024];
    let Ok(n) = stream.read(&mut buffer).await else {
        return;
    };

    if let Ok(data_from_client) = rmp_serde::from_slice::<CmdMessage>(&buffer[..n]) {
        if let Ok(mapfun_read) = mapfun.try_read() {
            if let Some(myfun) = mapfun_read.get(&data_from_client.cmd) {
                let cmd_response = myfun(data_from_client).await;

                if let Ok(buf) = rmp_serde::to_vec_named(&cmd_response) {
                    _ = stream.write_all(&buf).await;
                }
            }
        }
    };
    _ = stream.flush().await;
}
pub type MyBoxFuture = Pin<Box<dyn Future<Output = CmdMessage> + Send + 'static>>;
pub type HashMapCallback = HashMap<String, fn(CmdMessage) -> MyBoxFuture>;
pub type ArcCallBack = Arc<RwLock<HashMapCallback>>;

pub struct IpcRouter {
    pub callbacks: ArcCallBack,
    pub appctx: AppMyContext,
}

impl IpcRouter {
    pub fn new(appctx: AppMyContext) -> Self {
        Self {
            callbacks: Arc::new(RwLock::new(HashMap::new())),
            appctx,
        }
    }

    pub fn insert_callback(&mut self, cmd: &str, func: fn(CmdMessage) -> MyBoxFuture) {
        let Ok(mut lock) = self.callbacks.try_write() else {
            return;
        };

        lock.insert(cmd.to_string(), func);
    }

    pub async fn create_ipc_server(&self) {
        let ipcname = &self.appctx.ipc_name;
        let Ok(server_name) = ipcname.to_string().to_ns_name::<GenericNamespaced>() else {
            return;
        };

        let Ok(listener) = ListenerOptions::new().name(server_name).create_tokio() else {
            return;
        };

        loop {
            let Ok(stream) = listener.accept().await else {
                continue;
            };

            let mapfn = self.callbacks.clone();
            tokio::spawn(testfn(stream,mapfn));
        }
    }
}
